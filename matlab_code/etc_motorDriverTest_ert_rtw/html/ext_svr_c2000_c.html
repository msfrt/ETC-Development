151/* Function: GrowRecvBufIfNeeded ===============================================



152 * Abstract:



153 *  Allocate or increase the size of buffer for receiving packets from target.



154 */



155PRIVATE boolean_T GrowRecvBufIfNeeded(const int pktSize)



156{



157    if (pktSize > pktBufSize) {



158        if (pktBuf != NULL) {



159            free(pktBuf);



160            pktBufSize = 0;



161        }



162



163        pktBuf = (char *)malloc(pktSize);



164        if (pktBuf == NULL) return(EXT_ERROR);



165



166        pktBufSize = pktSize;



167    }



168    return(EXT_NO_ERROR);



169} /* end GrowRecvBufIfNeeded */

171/* Function: GetPktHdr =========================================================



172 * Abstract:



173 *  Attempts to retrieve a packet header from the host.  If a header is in



174 *  fact retrieved, the reference arg, 'hdrAvail' will be returned as true.



175 *



176 *  EXT_NO_ERROR is returned on success, EXT_ERROR is returned on failure.



177 *



178 * NOTES:



179 *  o It is not necessarily an error for 'hdrAvail' to be returned as false.



180 *    It typically means that we were polling for packets and none were



181 *    available.



182 */



183PRIVATE boolean_T GetPktHdr(PktHeader *pktHdr, boolean_T *hdrAvail)



184{



185    int_T     nGot      = 0; /* assume */



186    int_T     nGotTotal = 0;



187    int_T     pktSize   = sizeof(PktHeader)*SIZE_ADJUST;			/* Size Adjust for access of PktHeader size as per Host */



188    boolean_T error     = EXT_NO_ERROR;



189    char Temp_Buf[8];												/* Temp Buffer to Collect Header Data from Host*/



190



191    /* Get the header. */



192    while(nGotTotal < pktSize) {



193        error = ExtGetHostPkt(extUD,



194            pktSize - nGotTotal, &nGot, (char_T *)(Temp_Buf + nGotTotal));



195        if (error) goto EXIT_POINT;



196



197		BytePacker32bits(&pktHdr->type, &Temp_Buf[0]);		/* Pack Data from Temp_Buf into pktHdr */



198		BytePacker32bits(&pktHdr->size, &Temp_Buf[4]);



199



200



201



202	nGotTotal += nGot;



203



204        if (nGotTotal == 0) break;



205    }



206    assert((nGot == 0) || (nGotTotal == pktSize));



207



208EXIT_POINT:



209    *hdrAvail = (boolean_T)(nGot > 0);



210    return(error);



211} /* end GetPktHdr */

214/* Function: GetPktHdrUsingChecksum =========================================================



215 *



216 */



217PRIVATE boolean_T GetPktHdrUsingChecksum(PktHeader *pktHdr, boolean_T *hdrAvail)



218{



219    int_T     nGot      = 0; /* assume */



220    int_T     nGotTotal = 0;



221    int_T     pktSize   = sizeof(PktHeader)*SIZE_ADJUST;



222    boolean_T error     = EXT_NO_ERROR;



223	



224	int32_T retries = 0;



225	



226	static char bufr[10];



227	int_T bufferSize = pktSize + 2;



228	



229    /* Get the header. */



230    while(nGotTotal < bufferSize) {



231        error = ExtGetHostPkt(extUD,



232            1, &nGot, (char_T*)((char_T*)bufr + nGotTotal));



233        if (error) goto EXIT_POINT;



234



235		nGotTotal += nGot;



236		if(nGotTotal == 0)



237			break;



238		



239		retries += 1;



240		



241		if(retries >= MAX_RETRIES){



242			SendPktHdrToHost((ExtModeAction)0x000000ED, seqId);



243			nGot = 0;



244			break;



245		}



246		



247		if(nGotTotal == bufferSize){



248			if(verify_crc8_j1850(bufr, 0, 9)){



249				if((seqId == (uint8_T)bufr[pktSize])){



250					seqId = (uint8_T)bufr[pktSize];



251					SendPktHdrToHost((ExtModeAction)0x000000EE, seqId);



252					BytePacker32bits(&pktHdr->type, &bufr[0]);	BytePacker32bits(&pktHdr->size, &bufr[4]);



253					seqId+=1;	seqId&=0x00FF;



254				} else {



255					nGotTotal = 0;



256					nGot = 0;



257				}



258			} else {



259				SendPktHdrToHost((ExtModeAction)0x000000EF, seqId);



260				nGot = 0;



261			}



262		}



263	}



264	



265    //assert((nGot == 0) || (nGotTotal == pktSize));



266



267EXIT_POINT:



268    *hdrAvail = (boolean_T)(nGot > 0);



269    return(error);



270} /* end GetPktHdrUsingChecksum */

273/* Function: ClearPkt ==========================================================



274 * Abstract:



275 *  Remove the data from the communication line one byte at a time.  This



276 *  function is called when there was not enough memory to receive an entire



277 *  packet.  Since the data was never received, it must be discarded so that



278 *  other packets can be sent.



279 */



280PRIVATE void ClearPkt(const int pktSize)



281{



282    int_T     nGot;



283    boolean_T error     = EXT_NO_ERROR;



284    int_T     nGotTotal = 0;



285    static    char buffer;



286



287    /* Get and discard the data one char at a time. */



288    while(nGotTotal < pktSize) {



289        error = ExtGetHostPkt(extUD, 1, &nGot, (char_T *)&buffer);



290        if (error) {



291#ifndef EXTMODE_DISABLEPRINTF



292	    fprintf(stderr,"ExtGetHostPkt() failed.\n");



293#endif



294            goto EXIT_POINT;



295	}



296



297	nGotTotal += nGot;



298    }



299



300EXIT_POINT:



301    return;



302



303} /* end ClearPkt */

306/* Function: GetPkt ============================================================



307 * Abstract:



308 *  Receive nBytes from the host.  Return a buffer containing the bytes or



309 *  NULL if an error occurs.  Note that the pointer returned is that of the



310 *  global pktBuf.  If the buf needs to be grown to accommodate the package,



311 *  it is realloc'd.  This function will try to get the requested number



312 *  of bytes indefinitely - it is assumed that the data is either already there,



313 *  or will show up in a "reasonable" amount of time.



314 */



315PRIVATE const char *GetPkt(const int pktSize)



316{



317    int_T     nGot;



318    boolean_T error     = EXT_NO_ERROR;



319    int_T     nGotTotal = 0;



320



321    error = GrowRecvBufIfNeeded(pktSize);



322    if (error != EXT_NO_ERROR) {



323#ifndef EXTMODE_DISABLEPRINTF



324        fprintf(stderr,"Previous pkt from host thrown away due to lack of memory.\n");



325#endif



326        ClearPkt(pktSize);



327        goto EXIT_POINT;



328    }



329



330    /* Get the data. */



331    while(nGotTotal < pktSize) {



332        error = ExtGetHostPkt(extUD,



333            pktSize - nGotTotal, &nGot, (char_T *)(pktBuf + nGotTotal));



334        if (error) {



335#ifndef EXTMODE_DISABLEPRINTF



336	    fprintf(stderr,"ExtGetHostPkt() failed.\n");



337#endif



338            goto EXIT_POINT;



339	}



340



341	nGotTotal += nGot;



342    }



343



344EXIT_POINT:



345    return((error == EXT_NO_ERROR) ? pktBuf : NULL);



346} /* end GetPkt */

354/* Function: DisconnectFromHost ================================================



355 * Abstract:



356 *  Disconnect from the host.



357 */



358PRIVATE void DisconnectFromHost(int_T numSampTimes)



359{



360    int i;



361



362    for (i=0; i<NUM_UPINFOS; i++) {



363        UploadPrepareForFinalFlush(i);



364



365#if defined(VXWORKS)



366        /*



367         * UploadPrepareForFinalFlush() has already called semGive(uploadSem)



368         * two times.  Now the server thread will wait until the upload thread



369         * has processed all of the data in the buffers for the final upload



370         * and exhausted the uploadSem semaphores.  If the server thread



371         * attempts to call UploadServerWork() while the upload thread is in



372         * the middle of processing the buffers, the target code may crash



373         * with a NULL pointer exception (the buffers are destroyed after



374         * calling UploadLogInfoTerm).



375         */



376        while(semTake(uploadSem, NO_WAIT) != ERROR) {



377            semGive(uploadSem);



378            taskDelay(1000);



379        }



380#else



381#ifndef EXTMODE_DISABLESIGNALMONITORING



382        if (host_upstatus_is_uploading) {



383            UploadServerWork(i, numSampTimes);



384        }



385#endif



386#endif



387



388        UploadLogInfoTerm(i, numSampTimes);



389    }



390



391    connected       = FALSE;



392    commInitialized = FALSE;



393	seqId			= 0;



394



395    ExtCloseConnection(extUD);



396} /* end DisconnectFromHost */

421/* Function: ProcessConnectPkt =================================================



422 * Abstract:



423 *  Process the EXT_CONNECT packet and send response to host.



424 */



425PRIVATE boolean_T ProcessConnectPkt(RTWExtModeInfo *ei)



426{



427    int_T                   nSet;



428    PktHeader               pktHdr;



429    int_T                   tmpBufSize;



430    uint32_T                *tmpBuf = NULL;



431    boolean_T               error   = EXT_NO_ERROR;



432



433    const DataTypeTransInfo *dtInfo    = rteiGetModelMappingInfo(ei);



434    uint_T                  *dtSizes   = dtGetDataTypeSizes(dtInfo);



435    int_T                   nDataTypes = dtGetNumDataTypes(dtInfo);



436	



437	uint8_T checksum;



438



439    assert(connected);



440    assert(!comminitialized);



441



442    /*



443     * Send the 1st of two EXT_CONNECT_RESPONSE packets to the host.



444     * The packet consists purely of the pktHeader.  In this special



445     * case the pktSize actually contains the number of bits per byte



446     * (not always 8 - see TI compiler for C30 and C40).



447     */



448    pktHdr.type = (uint32_T)EXT_CONNECT_RESPONSE;



449    pktHdr.size = (uint32_T)8; /*



450    							 mimicking the host that target is a 8 bit byte addressable unit



451							   */



452



453    error = ExtSetHostPkt(extUD,sizeof(pktHdr),(char_T *)&pktHdr,&nSet);



454    if (error || (nSet != sizeof(pktHdr))) {



455#ifndef EXTMODE_DISABLEPRINTF



456        fprintf(stderr,



457            "ExtSetHostPkt() failed for 1st EXT_CONNECT_RESPONSE.\n");



458#endif



459        goto EXIT_POINT;



460    }



461



462    /* Send 2nd EXT_CONNECT_RESPONSE packet containing the following



463     * fields:



464     *



465     * CS1 - checksum 1 (uint32_T)



466     * CS2 - checksum 2 (uint32_T)



467     * CS3 - checksum 3 (uint32_T)



468     * CS4 - checksum 4 (uint32_T)



469     *



470     * intCodeOnly   - flag indicating if target is integer only (uint32_T)



471     *



472     * MWChunkSize   - multiword data type chunk size on target (uint32_T)



473     *



474     * targetStatus  - the status of the target (uint32_T)



475     *



476     * nDataTypes    - # of data types        (uint32_T)



477     * dataTypeSizes - 1 per nDataTypes       (uint32_T[])



478     */



479



480    {



481        int nPktEls    = 4 +                        /* checkSums       */



482                         1 +                        /* intCodeOnly     */



483                         1 +                        /* MW chunk size   */



484                         1 +                        /* targetStatus    */



485                         1 +                        /* nDataTypes      */



486                         dtGetNumDataTypes(dtInfo); /* data type sizes */



487



488        tmpBufSize = nPktEls * sizeof(uint32_T) * SIZE_ADJUST;	/* Size Indicator to the host */



489        tmpBuf     = (uint32_T *)malloc(tmpBufSize);



490        if (tmpBuf == NULL) {



491            error = EXT_ERROR; goto EXIT_POINT;



492        }



493    }



494



495    /* Send packet header. */



496    pktHdr.type = EXT_CONNECT_RESPONSE;



497    pktHdr.size = tmpBufSize;



498



499    error = ExtSetHostPkt(extUD,sizeof(pktHdr),(char_T *)&pktHdr,&nSet);



500 if (error || (nSet != sizeof(pktHdr))) {



501#ifndef EXTMODE_DISABLEPRINTF



502        fprintf(stderr,



503            "ExtSetHostPkt() failed for 2nd EXT_CONNECT_RESPONSE.\n");



504#endif



505        goto EXIT_POINT;



506    }



507



508    /* Checksums, target status & SL_DOUBLESize. */



509    tmpBuf[0] = rteiGetChecksum0(ei);



510    tmpBuf[1] = rteiGetChecksum1(ei);



511    tmpBuf[2] = rteiGetChecksum2(ei);



512    tmpBuf[3] = rteiGetChecksum3(ei);



513



514#if INTEGER_CODE == 0



515    tmpBuf[4] = (uint32_T)0;



516#else



517    tmpBuf[4] = (uint32_T)1;



518#endif



519



520    tmpBuf[5] = (uint32_T)(SIZE_ADJUST * sizeof(uchunk_T));  /* Size Indicator to the host */



521



522    tmpBuf[6] = (uint32_T)modelStatus;



523



524    /* nDataTypes and dataTypeSizes */



525    {



526        int i;



527        tmpBuf[7] = (uint32_T)nDataTypes;



528        for (i=0; i<nDataTypes; i++) {



529            tmpBuf[8+i] = (uint32_T)SIZE_ADJUST*dtSizes[i];	  /* Mimicking DataTypes to be of size of a 8bit Byte Addressable Target */



530        }



531    }



532



533    /* Send the packet. */



534	tmpBufSize /= 2; 	/*



535							As the 16bit is split as 8bit characters in the lower layer



536							Data added to the packet will be in terms of target bytes



537						*/



538    error = ExtSetHostPkt(extUD,tmpBufSize,(char_T *)tmpBuf,&nSet);



539    if (error || (nSet != tmpBufSize)) {



540#ifndef EXTMODE_DISABLEPRINTF



541        fprintf(stderr,



542            "ExtSetHostPkt() failed.\n");



543#endif



544        goto EXIT_POINT;



545    }



546



547	/* Send the checksum. */



548	checksum = calculate_crc8_j1850_bytePacked((uint8_T *)tmpBuf, 0, tmpBufSize*SIZE_ADJUST);



549	error = ExtSetHostPkt8bit(extUD, 1, (char_T *)&checksum, &nSet);



550	



551    commInitialized = TRUE;



552



553EXIT_POINT:



554    free(tmpBuf);



555    return(error);



556} /* end ProcessConnectPkt */

588/* Function: ClearPktUsingAck ==================================================



589 * Abstract:



590 *  Remove the data from the communication line one byte at a time.  This



591 *  function is called when there was not enough memory to receive an entire



592 *  packet.  Since the data was never received, it must be discarded so that



593 *  other packets can be sent.



594 */



595PRIVATE void ClearPktUsingAck(const int pktSize)



596{



597    int_T     nGot;



598    boolean_T error     = EXT_NO_ERROR;



599    int_T     nGotTotal = 0;



600	



601	int_T     i;



602	int32_T   retries = 0;



603	int_T     nChunks   = pktSize/CHUNK_SIZE;



604	int_T     bytesLeft = pktSize%CHUNK_SIZE;



605	



606    /* Get, verify and discard chunks at a time. */



607    for(i=0; i<nChunks;){



608		retries = 0;



609		nGotTotal = 0;



610		while(nGotTotal < (CHUNK_SIZE+2)){



611			error = ExtGetHostPkt(extUD,



612						1,



613						&nGot,



614						(char_T *)(chunkBuffer + nGotTotal));



615			if (error) {



616#ifndef EXTMODE_DISABLEPRINTF                



617				fprintf(stderr,"ExtGetHostPkt() failed.\n");



618#endif



619            goto EXIT_POINT;



620			}



621			nGotTotal += nGot;



622			



623			retries++;



624			if(retries >= MAX_RETRIES){



625				SendPktHdrToHost((ExtModeAction)0x000000ED, seqId);



626				retries = 0;	nGotTotal = 0;



627			}



628			



629			if(nGotTotal == (CHUNK_SIZE+2)){



630				if(verify_crc8_j1850((uint8_T *)chunkBuffer, 0, CHUNK_SIZE+1)){



631						if(seqId == (uint8_T)chunkBuffer[CHUNK_SIZE]){



632							SendPktHdrToHost((ExtModeAction)0x000000EE, seqId);



633							seqId++;	seqId&=0x00FF;



634							i++;



635						}



636				}



637				else{



638					SendPktHdrToHost((ExtModeAction)0x000000EF, seqId);



639					retries = 0;



640					nGotTotal = 0;



641				}



642			}



643		}



644	}



645	



646	/*



647	 * Receive the bytes left over and reply with an ack after you receive all



648	 */



649	nGotTotal=0;



650	retries = 0;



651	if(bytesLeft != 0){



652		while(nGotTotal<(CHUNK_SIZE+2)){



653			error = ExtGetHostPkt(extUD,



654						1,



655						&nGot,



656						(char_T *)(chunkBuffer + nGotTotal));



657			if (error) {



658#ifndef EXTMODE_DISABLEPRINTF                



659				fprintf(stderr,"ExtGetHostPkt() failed.\n");



660#endif



661				goto EXIT_POINT;



662			}



663			nGotTotal += nGot;



664		



665			retries++;



666			if(retries >= MAX_RETRIES){



667				SendPktHdrToHost((ExtModeAction)0x000000ED, seqId);



668				retries = 0;	nGotTotal = 0;



669			}



670			



671			if(nGotTotal == (CHUNK_SIZE+2)){



672				if(verify_crc8_j1850((uint8_T *)chunkBuffer, 0, CHUNK_SIZE+1)){



673					if(seqId == (uint8_T)chunkBuffer[CHUNK_SIZE]){



674						SendPktHdrToHost((ExtModeAction)0x000000EE, seqId);



675						seqId++;	seqId&=0x00FF;



676					}



677				}



678				else{



679					SendPktHdrToHost((ExtModeAction)0x000000EF, seqId);



680					retries = 0;



681					nGotTotal = 0;



682				}



683			}



684		}

693/* Function: GetPktUsingAck ============================================================



694 * Abstract:



695 *  Receive nBytes from the host.  Return a buffer containing the bytes or



696 *  NULL if an error occurs.  Note that the pointer returned is that of the



697 *  global pktBuf.  If the buf needs to be grown to accommodate the package,



698 *  it is realloc'd.  This function will try to get the requested number



699 *  of bytes indefinitely - it is assumed that the data is either already there,



700 *  or will show up in a "reasonable" amount of time.



701 */



702



703PRIVATE const char *GetPktUsingAck(const int pktSize)



704{



705    int_T     nGot;



706    boolean_T error     = EXT_NO_ERROR;



707    int_T     nGotTotal = 0;



708	



709	int_T     i;



710	int32_T   retries = 0;



711	int_T     nChunks   = pktSize/CHUNK_SIZE;



712	int_T     bytesLeft = pktSize%CHUNK_SIZE;



713



714    error = GrowRecvBufIfNeeded(pktSize + 2);		// +2, checksum and seqID



715    if (error != EXT_NO_ERROR) {



716#ifndef EXTMODE_DISABLEPRINTF            



717        fprintf(stderr,"Previous pkt from host thrown away due to lack of memory.\n");



718#endif



719        ClearPktUsingAck(pktSize);



720        goto EXIT_POINT;



721    }



722	



723	/*



724	 * Receive chunks, verify and reply with ack



725	 */



726	for(i=0; i<nChunks;){



727		nGotTotal = 0;



728		retries = 0;



729		while(nGotTotal < (CHUNK_SIZE+2) ){			// 1B checksum and 1B seqID / chunk



730			error = ExtGetHostPkt(extUD,



731						1,



732						&nGot,



733						(char_T *)(pktBuf + i*CHUNK_SIZE + nGotTotal));



734			if (error) {



735#ifndef EXTMODE_DISABLEPRINTF                



736				fprintf(stderr,"ExtGetHostPkt() failed.\n");



737#endif



738				goto EXIT_POINT;



739			}



740			nGotTotal += nGot;



741			



742			retries++;



743			if(retries >= MAX_RETRIES){



744				SendPktHdrToHost((ExtModeAction)0x000000ED, seqId);



745				retries = 0;



746				nGotTotal = 0;



747			}



748			



749			/*



750			 * Do not reduce any of the nested if statements here



751			 */



752			if(nGotTotal == (CHUNK_SIZE+2)){



753				if(verify_crc8_j1850((uint8_T *)pktBuf, i*CHUNK_SIZE, CHUNK_SIZE+1)){



754					if(seqId == (uint8_T)pktBuf[i*CHUNK_SIZE+CHUNK_SIZE]){



755						SendPktHdrToHost((ExtModeAction)0x000000EE, seqId);



756						seqId+=1;	seqId&=0x00FF;



757						i++;



758					}



759				}



760				else{



761					SendPktHdrToHost((ExtModeAction)0x000000EF, seqId);



762					retries = 0;



763					nGotTotal = 0;



764				}



765			}



766		}



767	}



768	



769	//Receive the bytes left over and reply with an ack after you receive all



770	nGotTotal = 0;



771	retries = 0;



772	if(bytesLeft != 0){



773		while(nGotTotal<(CHUNK_SIZE+2)){



774			error = ExtGetHostPkt(extUD,



775						1,



776						&nGot,



777						(char_T *)(chunkBuffer + nGotTotal));



778			if (error) {



779#ifndef EXTMODE_DISABLEPRINTF                



780				fprintf(stderr,"ExtGetHostPkt() failed.\n");



781#endif



782				goto EXIT_POINT;



783			}



784			nGotTotal += nGot;



785			



786			retries++;



787			if(retries >= MAX_RETRIES){



788				SendPktHdrToHost((ExtModeAction)0x000000ED, seqId);



789				retries = 0;



790				nGotTotal = 0;



791			}



792			



793			/*



794			 * Do not reduce any of the nested if statements here



795			 */



796			if(nGotTotal == (CHUNK_SIZE+2)){



797				if(verify_crc8_j1850((uint8_T *)chunkBuffer, 0, CHUNK_SIZE+1)){



798					if(seqId == (uint8_T)chunkBuffer[CHUNK_SIZE]){



799						SendPktHdrToHost((ExtModeAction)0x000000EE, seqId);



800						memcpy(pktBuf + i*CHUNK_SIZE, chunkBuffer, bytesLeft);



801						seqId+=1;	seqId&=0x00FF;



802					}



803				}



804				else{



805					SendPktHdrToHost((ExtModeAction)0x000000EF, seqId);



806					retries = 0;



807					nGotTotal = 0;



808				}



809			}



810		}



811	}



812



813EXIT_POINT:



814    return((error == EXT_NO_ERROR) ? pktBuf : NULL);



815} /* end GetPktUsingAck */

1218/* Function: UploadServerWork =================================================



1219 * Abstract:



1220 *  Upload model signals to host for a single upInfo.



1221 */



1222void UploadServerWork(int32_T upInfoIdx, int_T numSampTimes)



1223{



1224    int_T         i;



1225    ExtBufMemList upList;



1226    boolean_T     error = EXT_NO_ERROR;



1227	



1228	const int8_T headerLength = 8/SIZE_ADJUST;



1229	uint8_T checksum;



1230	uint32_T bytesInUploadBuffer, pktHdrSize, bytesSent;



1231	



1232	int_T nSet;



1233	static int_T buffer1Length, buffer2Length;



1234	static char_T *buffer1, *buffer2;



1235



1236#ifdef VXWORKS



1237    /*



1238     * Don't spin the CPU unless we've got data to upload.



1239     * The upload.c/UploadBufAddTimePoint function gives the sem



1240     * each time that data is added.



1241     */



1242taskUnsafe();



1243    semTake(uploadSem, WAIT_FOREVER);



1244taskSafe();



1245#endif



1246



1247    if (!connected) goto EXIT_POINT;



1248



1249    UploadBufGetData(&upList, upInfoIdx, numSampTimes);



1250    while(upList.nActiveBufs > 0) {

1252            const BufMem *bufMem = &upList.bufs[i];



1253			



1254			buffer1Length = bufMem->nBytes1;



1255			buffer2Length = bufMem->nBytes2;



1256			buffer1 = bufMem->section1;



1257			buffer2 = bufMem->section2;



1258			bytesInUploadBuffer = buffer1Length + buffer2Length;



1259			



1260			/*



1261			 * External mode packets consist of 8B headers followed by variable size payloads.



1262			 * These packets are stored in two non-contiguous buffers, section1 and section2.



1263			 * A packet may be split between the buffers.



1264			 * The size of each is given by size1 and size2 respectively.



1265			 */



1266			



1267			bytesSent = 0;



1268			while(bytesSent < bytesInUploadBuffer){



1269				pktHdrSize = 0;



1270				checksum = 0;



1271				/* we fetch the words indivudually as we do not know if they all are contiguous in section1 or section2 */



1272				pktHdrSize |= (uint32_T)(*LOC(bytesSent+2));



1273				pktHdrSize |= (uint32_T)(*LOC(bytesSent+3))<<16;



1274				pktHdrSize = pktHdrSize/SIZE_ADJUST;



1275				



1276				if( (bytesSent + headerLength) <= buffer1Length ){ 					/* check if the next header is split between the buffers */

1283								/*calcualte checksum in one go and send it off as well*/



1284						checksum = calculate_crc8_j1850_bytePacked((uint8_T *)LOC(bytesSent), 0, (uint16_T)(pktHdrSize)*SIZE_ADJUST);

1288							/* send the remaining payload in section2 */



1289						error = SendPktDataToHost(buffer2, (uint16_T)(bytesSent + pktHdrSize - buffer1Length));



1290							/* calculate the checksum for the payload in section1 */



1291						checksum = calculate_crc8_j1850_bytePacked((uint8_T *)LOC(bytesSent), 0, (uint16_T)(buffer1Length - bytesSent)*SIZE_ADJUST);



1292							/* calculate the checksum for the payload in section2 */



1293						checksum = calculate_crc8_j1850_withInit_bytePacked((uint8_T *)buffer2, checksum, 0, (uint16_T)(bytesSent + pktHdrSize - buffer1Length)*SIZE_ADJUST);



1294					}



1295					bytesSent+=pktHdrSize;



1296				} else {															/* land here if header is split OR is in section2.



1297																					 * also guarantees that the entire payload for this



1298																					 * header is in section2.



1299																					 */



1300					if(bytesSent < buffer1Length){		/* if header is split */



1301						error = SendPktDataToHost(LOC(bytesSent), buffer1Length - bytesSent);



1302						error = SendPktDataToHost(buffer2, (headerLength + pktHdrSize - (buffer1Length - bytesSent)));



1303					} else {								/* the entire packet is in section2 */



1304						error = SendPktDataToHost(LOC(bytesSent), headerLength + pktHdrSize);



1305					}



1306					checksum = calculate_crc8_j1850_bytePacked((uint8_T *)LOC(bytesSent + headerLength), 0, (uint16_T)(pktHdrSize)*SIZE_ADJUST);



1307					bytesSent += (headerLength + pktHdrSize);



1308				}



1309				/* finally send the checksum */



1310				error = ExtSetHostPkt8bit(extUD, 1, (char_T *)&checksum, &nSet);



1311			}



1312			



1313            /* confirm that the data was sent */



1314            UploadBufDataSent(upList.tids[i], upInfoIdx);



1315        }



1316		



1317        UploadBufGetData(&upList, upInfoIdx, numSampTimes);



1318    }



1319



1320



1321EXIT_POINT:



1322    if (error != EXT_NO_ERROR) {



1323        /* An error in this function is caused by a physical failure in the



1324         * external mode connection.  We assume this failure caused the host



1325         * to disconnect.  The target must be disconnected and returned to a



1326         * state where it is running and can be re-connected to by the host.



1327         */



1328        ForceDisconnectFromHost(numSampTimes);



1329    }



1330}

1414     * If not connected, attempt to make connection to host.



1415     */

1417        rtExtModeTestingKillIfOrphaned(FALSE);



1418

1424     * If ExtOpenConnection is not blocking and there are no pending



1425     * requests to open a connection, we'll still be unconnected.



1426     */

1428

1430     * Process packets.



1431     */



1432

1436	else

1445    }



1446    rtExtModeTestingKillIfOrphaned(hdrAvail);



1447

1449

1451     * This is the first packet.  Should contain the string:



1452     * 'ext-mode'.  Its contents are not important to us.



1453     * It is used as a flag to start the handshaking process.

1455    if (!commInitialized) {



1456        pktHdr.type = EXT_CONNECT;



1457    }



1458



1459    /*



1460     * At this point we know that we have a packet: process it.



1461     */



1462#ifdef VXWORKS



1463    taskSafe();



1464#endif



1465    switch(pktHdr.type) {



1466



1467    case EXT_GET_TIME:



1468    {



1469        /* Skip verbosity print out - we get too many of these */



1470        /*PRINT_VERBOSE(("got EXT_GET_TIME packet.\n"));*/



1471        time_T t = rteiGetT(ei);



1472        error = SendPktToHost(



1473            EXT_GET_TIME_RESPONSE,sizeof(time_T),

1476        break;



1477    }



1478



1479    case EXT_ARM_TRIGGER:

1481#ifndef EXTMODE_DISABLESIGNALMONITORING

1484#else

1486        if (error != EXT_NO_ERROR) goto EXIT_POINT;



1487#endif



1488        break;



1489    }



1490



1491    case EXT_SELECT_SIGNALS:



1492    {



1493#ifndef EXTMODE_DISABLESIGNALMONITORING



1494        error = ProcessSelectTriggerSignalPkt(EXT_SELECT_SIGNALS_RESPONSE, ei, pktHdr.size, numSampTimes, ERRMSG_PROCESSSELECTSIGNAL);

1498        if (error != EXT_NO_ERROR) goto EXIT_POINT;



1499#endif



1500        break;



1501    }



1502



1503    case EXT_SELECT_TRIGGER:



1504    {



1505#ifndef EXTMODE_DISABLESIGNALMONITORING

1511#endif



1512        break;



1513    }



1514



1515    case EXT_CONNECT:



1516    {



1517        PRINT_VERBOSE(("got EXT_CONNECT packet.\n"));



1518        error = ProcessConnectPkt(ei);



1519        if (error != EXT_NO_ERROR) goto EXIT_POINT;



1520        break;



1521    }



1522



1523    case EXT_SETPARAM:



1524    {



1525#ifndef EXTMODE_DISABLEPARAMETERTUNING

1527        error = ProcessSetParamPkt(ei, pktHdr.size);

1533#endif

1629



1630    case EXT_CANCEL_LOGGING:



1631    {



1632#ifndef EXTMODE_DISABLESIGNALMONITORING



1633        error = ProcessCancelLoggingArmTriggerPkt(EXT_CANCEL_LOGGING_RESPONSE, pktHdr.size, numSampTimes);
