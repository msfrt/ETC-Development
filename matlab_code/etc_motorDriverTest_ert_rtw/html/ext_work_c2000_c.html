38void rtExtModeTornadoStartup(RTWExtModeInfo *ei,



39                             int_T          numSampTimes,



40                             boolean_T      *stopReqPtr,



41                             int_T          priority,



42                             int32_T        stack_size,



43                             SEM_ID         startStopSem)



44{



45    uploadSem = semBCreate(SEM_Q_PRIORITY, SEM_EMPTY);



46    commSem   = semBCreate(SEM_Q_PRIORITY, SEM_FULL);



47    pktSem    = semBCreate(SEM_Q_PRIORITY, SEM_FULL);



48



49    rt_ExtModeInit();



50



51    extern_pkt_tid = taskSpawn("tExternPkt",



52        priority+(numSampTimes), VX_FP_TASK, stack_size, (FUNCPTR)rt_PktServer,



53        (int_T) ei, (int_T) numSampTimes, (int_T) stopReqPtr, 0, 0, 0, 0, 0, 0, 0);



54    if (extern_pkt_tid == ERROR) {



55#ifndef EXTMODE_DISABLEPRINTF



56        printf("handle taskpawn error");



57#endif



58    }



59



60    extern_upload_tid = taskSpawn("tExternUpload",



61        priority+(numSampTimes)+1,VX_FP_TASK, stack_size,(FUNCPTR)rt_UploadServer,



62        (int_T) numSampTimes, 0, 0, 0, 0, 0, 0, 0, 0, 0);



63    if (extern_upload_tid == ERROR) {



64#ifndef EXTMODE_DISABLEPRINTF



65        printf("handle taskpawn error");



66#endif



67    }



68



69    /*



70     * Pause until receive model start packet - if external mode.



71     * Make sure the external mode tasks are running so that



72     * we are listening for commands from the host.



73     */



74    if (ExtWaitForStartPkt()) {



75#ifndef EXTMODE_DISABLEPRINTF



76        printf("\nWaiting for start packet from host.\n");



77#endif



78        semTake(startStopSem, WAIT_FOREVER);



79    }



80    modelStatus = TARGET_STATUS_RUNNING;



81}

86    taskDelete(extern_pkt_tid);

88    semDelete(uploadSem);



89    semDelete(commSem);



90    semDelete(pktSem);



91}

108/* Function ====================================================================



109 * Pause the process (w/o hogging the cpu) until receive step packet (which



110 * means the startModel flag moves to true) or until we are no longer



111 * in the paused state.  The packet/upload server must continue to process



112 * events (otherwise the host would not be able to communicate with the target).



113 */



114void rtExtModePauseIfNeeded(RTWExtModeInfo *ei,



115                            int_T          numSampTimes,



116                            boolean_T      *stopReqPtr)



117{



118    while((modelStatus == TARGET_STATUS_PAUSED) &&



119          !startModel && !(*stopReqPtr)) {



120        rt_ExtModeSleep(0L, 375000L);



121        rt_PktServerWork(ei,numSampTimes,stopReqPtr);



122#ifndef EXTMODE_DISABLESIGNALMONITORING



123        rt_UploadServerWork(numSampTimes);



124#endif



125    }



126    startModel = FALSE; /* reset to FALSE - if we were stepped we want to



127                         *                  stop again next time we get



128                         *                  back here.



129                         */



130} /* end rtExtModePauseIfNeeded */



131



132/* Function ====================================================================



133 * Pause the process (w/o hogging the cpu) until receive start packet



134 * from the host.  The packet/upload server must continue to process



135 * events (otherwise the host would not be able to communicate with the target).



136 */



137void rtExtModeWaitForStartPkt(RTWExtModeInfo *ei,



138                              int_T          numSampTimes,



139                              boolean_T      *stopReqPtr)



140{



141    /*



142     * Pause until receive model start packet.



143     */



144    if (ExtWaitForStartPkt()) {



145        while(!startModel && !(*stopReqPtr)) {



146            rt_ExtModeSleep(0L, 375000L);



147            rt_PktServerWork(ei,numSampTimes,stopReqPtr);



148#ifndef EXTMODE_DISABLESIGNALMONITORING



149            rt_UploadServerWork(numSampTimes);



150#endif



151        }



152    }



153    if (modelStatus != TARGET_STATUS_PAUSED) {



154        modelStatus = TARGET_STATUS_RUNNING;



155    } else {



156        /* leave in pause mode */



157    }



158}

219void rtExtModeParseArgs(int_T        argc,



220                        const char_T *argv[],



221                        real_T       *unused)



222{



223    UNUSED_PARAMETER(unused);



224



225#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



226    /* Some targets do not support command line args, so bypass the args



227     * processing and directly call into ExtParseArgsAndInitUD.



228     */



229



230    /* initialize external mode */



231    ExtParseArgsAndInitUD(0, NULL);



232



233#else



234



235    /*



236     * Parse the external mode arguments.



237     */



238    {



239        const char_T *extParseErrorPkt = ExtParseArgsAndInitUD(argc, argv);



240        if (extParseErrorPkt != NULL) {



241#ifndef EXTMODE_DISABLEPRINTF



242            printf(



243                "\nError processing External Mode command line arguments:\n");



244            printf("\t%s",extParseErrorPkt);



245#endif



246            exit(EXIT_FAILURE);



247        }



248    }



249#endif  /* EXTMODE_DISABLE_ARGS_PROCESSING */



250}

287void rtParseArgsForExtMode(int_T        argc,



288                           const char_T *argv[])



289{



290



291#ifdef  EXTMODE_DISABLE_ARGS_PROCESSING



292    /* Some targets do not support command line args, so bypass the args



293     * processing and directly call into ExtParseArgsAndInitUD.



294     */



295



296    /* initialize external mode */



297    ExtParseArgsAndInitUD(0, NULL);



298



299#else



300    /* parse command line args */



301



302    int_T  oldStyle_argc;



303    const char_T *oldStyle_argv[5];



304



305    if ((argc > 1) && (argv[1][0] != '-')) {



306        /* old style */



307        if ( argc > 3 ) {



308            displayUsage();



309            exit(EXIT_FAILURE);



310        }



311



312        oldStyle_argc    = 1;



313        oldStyle_argv[0] = argv[0];



314



315        if (argc >= 2) {



316            oldStyle_argc = 3;



317



318            oldStyle_argv[1] = "-tf";



319            oldStyle_argv[2] = argv[1];



320        }



321



322        if (argc == 3) {



323            oldStyle_argc = 5;



324



325            oldStyle_argv[3] = "-port";



326            oldStyle_argv[4] = argv[2];



327



328        }



329



330        argc = oldStyle_argc;



331        argv = oldStyle_argv;



332



333    }



334



335    {



336        /* new style: */



337        double    tmpDouble;



338        char_T tmpStr2[200];



339        int_T  count      = 1;



340        int_T  parseError = FALSE;



341



342        /*



343         * Parse the standard RTW parameters.  Let all unrecognized parameters



344         * pass through to external mode for parsing.  NULL out all args handled



345         * so that the external mode parsing can ignore them.



346         */



347        while(count < argc) {



348            const char_T *option = argv[count++];



349



350            /* final time */



351            if ((strcmp(option, "-tf") == 0) && (count != argc)) {



352                const char_T *tfStr = argv[count++];



353



354                sscanf(tfStr, "%200s", tmpStr2);



355                if (strcmp(tmpStr2, "inf") == 0) {



356                    tmpDouble = RUN_FOREVER;



357                } else {



358                    char_T tmpstr[2];



359



360#if INTEGER_CODE == 0



361                    if ( (sscanf(tmpStr2,"%lf%1s", &tmpDouble, tmpstr) != 1) ||



362                         (tmpDouble < (real_T)0) ) {



363    #ifndef EXTMODE_DISABLEPRINTF



364                        (void)printf("finaltime must be a positive, real value or inf\n");



365    #endif



366                        parseError = TRUE;



367                        break;



368                    }



369#else



370                    if ( (sscanf(tmpStr2,"%d%1s", &tmpDouble, tmpstr) != 1) ||



371                         (tmpDouble < (real_T)0) ) {



372    #ifndef EXTMODE_DISABLEPRINTF



373                        (void)printf("tmpDouble = %d\n", tmpDouble);



374                        (void)printf("finaltime must be a positive, integer value or inf\n");



375    #endif



376                        parseError = TRUE;



377                        break;



378                    }



379#endif



380                }



381                finaltime = (real_T) tmpDouble;



382



383                argv[count-2] = NULL;



384                argv[count-1] = NULL;



385            }



386        }



387



388        if (parseError) {



389    #ifndef EXTMODE_DISABLEPRINTF



390            (void)printf("\nUsage: model_name -option1 val1 -option2 val2 -option3 "



391                         "...\n\n");



392            (void)printf("\t-tf 20 - sets final time to 20 seconds\n");



393    #endif



394



395            exit(EXIT_FAILURE);



396        }



397



398        /*



399         * Parse the external mode arguments.



400         */



401        {



402            const char_T *extParseErrorPkt = ExtParseArgsAndInitUD(argc, argv);



403            if (extParseErrorPkt != NULL) {



404    #ifndef EXTMODE_DISABLEPRINTF



405                printf(



406                    "\nError processing External Mode command line arguments:\n");



407                printf("\t%s",extParseErrorPkt);



408    #endif



409                exit(EXIT_FAILURE);



410            }



411        }



412



413        /*



414         * Check for unprocessed ("unhandled") args.



415         */



416        {



417            int i;



418            for (i=1; i<argc; i++) {



419                if (argv[i] != NULL) {



420    #ifndef EXTMODE_DISABLEPRINTF



421                    printf("Unexpected command line argument: %s\n",argv[i]);



422    #endif



423                    exit(EXIT_FAILURE);



424                }



425            }



426        }



427    }



428



429    if (finaltime == RUN_FOREVER) {



430        #ifndef EXTMODE_DISABLEPRINTF



431                printf ("\n**warning: the simulation will run with no stop time due "



432                "to external mode with '-tf inf' argument.\n");



433        #endif



434    }



435



436#endif /* EXTMODE_DISABLE_ARGS_PROCESSING */



437}
