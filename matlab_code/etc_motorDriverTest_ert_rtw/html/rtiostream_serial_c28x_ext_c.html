108interrupt void SCIRXINTA_isr(void)



109{



110  queRx.queElements[((queRx.firstElement + queRx.queCount++) & (QUE_SIZE-1))] = SciaRegs.SCIRXBUF.all;



111  /*



112  	 Assumption: queRx (SW Queue) will not over flow if the QUE_SIZE is high (64 default value)



113	 The queRx is used to collect data when the SCI HW FIFO overflows



114	 Otherwise the Data Collection for external Mode occurs by



115	 polling the SCI Rx Buffer



116  */



117



118  SciaRegs.SCIFFRX.bit.RXFFINTCLR = 1; /* Clear Interrupt flag*/



119  PieCtrlRegs.PIEACK.all = PIEACK_GROUP9;/* Acknowledge to receive more interrupts   */



120}

122void enable_interrupts_scia(void)



123{



124  EALLOW;



125  #ifdef F2837X_REG_FORMAT



126  PieVectTable.SCIA_RX_INT = &SCIRXINTA_isr;/* Hook interrupt to the ISR*/

129  #endif



130  EDIS;



131  PieCtrlRegs.PIEIER9.bit.INTx1 = 1;   /* Enable PIE group 9 interrupt 1 for SCIRXINTA*/



132  IER |= M_INT9;



133}

168int rtIOStreamSend(int streamID, const void *src, size_t size, size_t *sizeSent)



169{



170	size_t sizeCharsToSend;



171	int escapecharAdjust=0;



172



173	sizeCharsToSend = size*2;



174	*sizeSent = size;



175



176	if (sizeCharsToSend > RTIOSTREAM_MAX_CHAR) {



177		sizeCharsToSend = RTIOSTREAM_MAX_CHAR;



178	*sizeSent = RTIOSTREAM_MAX_CHAR/2;



179	}



180



181	byteswap_L8exp_esc(srcSerial, (char *) src, *sizeSent, 2, &escapecharAdjust);



182



183	sizeCharsToSend += escapecharAdjust;



184	scia_xmit_external_mode(srcSerial, sizeCharsToSend);



185



186	return RTIOSTREAM_NO_ERROR;

238void init_SCI_external_mode(void)



239{



240	EALLOW;



241    /* initialize SCI & FIFO registers */



242	/*



243	* Initialize SCI_A with following parameters:



244	*    BaudRate              : 115200



245	*    CharacterLengthBits   : 8



246	*    EnableLoopBack        : 0



247	*    NumberOfStopBits      : 1



248	*    ParityMode            : None



249	*    SuspensionMode        : Free_run



250	*    CommMode              : Raw_data



251	*/



252    #ifdef F2837X_REG_FORMAT



253    CpuSysRegs.PCLKCR7.bit.SCI_A = 1;



254    #endif



255	SciaRegs.SCICTL1.bit.SWRESET = 0;    //Software reset



256	SciaRegs.SCICTL1.bit.SWRESET = 1;    //Software reset



257	//SysCtrlRegs.LOSPCP.bit.LSPCLK = 0;	// default is 2 (SYSCLk/4) Antonin (Remove the change)



258	SciaRegs.SCICCR.bit.STOPBITS = 0;    //Number of stop bits. (0: One stop bit, 1: Two stop bits)



259	SciaRegs.SCICCR.bit.PARITY = 0;      //Parity mode (0: Odd parity, 1: Even parity)



260	SciaRegs.SCICCR.bit.PARITYENA = 0;   //Enable Parity Mode



261	SciaRegs.SCICCR.bit.LOOPBKENA = 0;   //Loop Back enable



262	SciaRegs.SCICCR.bit.ADDRIDLE_MODE = 0;//ADDR/IDLE Mode control



263	SciaRegs.SCICCR.bit.SCICHAR = 7;     //Character length



264	SciaRegs.SCICTL1.bit.RXERRINTENA = 0;//Disable receive error interrupt



265	//SciaRegs.SCICTL1.bit.SWRESET = 1;    //Software reset



266	SciaRegs.SCICTL1.bit.TXENA = 1;      // SCI transmitter enable



267	SciaRegs.SCICTL1.bit.RXENA = 1;      // SCI receiver enable



268    



269    #ifdef F2837X_REG_FORMAT



270    SciaRegs.SCIHBAUD.all = MW_PIL_SCIHBAUD;



271	SciaRegs.SCILBAUD.all = MW_PIL_SCILBAUD;



272    //Free run, continue SCI operation regardless of suspend



273	SciaRegs.SCIPRI.bit.FREESOFT = 2;        // Free emulation mode control



274    #else



275	SciaRegs.SCIHBAUD = MW_PIL_SCIHBAUD;



276	SciaRegs.SCILBAUD = MW_PIL_SCILBAUD;



277    //Free run, continue SCI operation regardless of suspend



278	SciaRegs.SCIPRI.bit.FREE = 1;        // Free emulation mode control



279	SciaRegs.SCIPRI.bit.SOFT = 0;        // Interrupt priority select



280    #endif



281



282	



283	SciaRegs.SCIFFCT.bit.ABDCLR = 0;



284	SciaRegs.SCIFFCT.bit.CDC = 0;



285	SciaRegs.SCIFFTX.bit.SCIRST = 1;     // SCI reset rx/tx channels



286	SciaRegs.SCIFFTX.bit.SCIFFENA = 1;   // SCI FIFO enhancements are enabled.



287	#ifdef F2837X_REG_FORMAT



288		SciaRegs.SCIFFTX.bit.TXFIFORESET = 1;// Re-enable transmit FIFO operation.



289	#else



290		SciaRegs.SCIFFTX.bit.TXFIFOXRESET = 1;// Re-enable transmit FIFO operation.



291	#endif



292	SciaRegs.SCIFFRX.bit.RXFIFORESET = 1;// Re-enable receive FIFO operation.



293



294



295    SciaRegs.SCIFFRX.bit.RXFFIENA = 1;					//Receive FIFO Interrupt Enable



296    SciaRegs.SCIFFRX.bit.RXFFIL = MW_PIL_SCIFIFOLEN;	//Generate Interrupt when FIFO Overflows



297	SciaRegs.SCICTL1.bit.RXERRINTENA = 1; // Enable receive error interrupt to handle any receive errors for SCI



298



299



300	EDIS;



301}

361/* Function: scia_rcv_external_mode ================================================



362 *



363 * Abstract:



364 *  Receive Data from the SCI-A module, or from the SW Queue that is filled by the



365    SCI-A module



366 */



367#ifdef DAEMON_MODE



368int scia_rcv_external_mode(char *rcvBuff, int buffLen, int loopMode, int *sizeCharsRcvd)



369{



370	int i;



371	int charsRcvd = 0;



372



373	for (i = 0; i < buffLen; i++) {



374



375		if(deleteElement(&queRx, &rcvBuff[charsRcvd]))  {		//try to pull off a byte from the queue.



376			charsRcvd += 1;



377		}



378



379		if( (SciaRegs.SCIRXST.bit.FE) || (SciaRegs.SCIRXST.bit.PE) || (SciaRegs.SCIFFRX.bit.RXFFOVF) || (SciaRegs.SCIRXST.bit.BRKDT))



380		{



381			return FRAMERR;					/* Providing FRAME error for frame error, parity error and FIFO overflow



382											 as the RTIOStream doesnt have capability to differentitate between errors */



383		}



384	}



385	*sizeCharsRcvd = charsRcvd;



386	return NOERROR;



387}



388#else



389int scia_rcv_external_mode(char *rcvBuff, int buffLen, int loopMode, int*sizeCharsRcvd)



390{



391	int i;



392



393	*sizeCharsRcvd = 0;



394



395	for (i = 0; i < buffLen; i++) {



396



397		while(!deleteElement(&queRx, &rcvBuff[i]))  {}		/* Wait Until Data Received */



398



399    	*sizeCharsRcvd = *sizeCharsRcvd + 1;



400



401		if( (SciaRegs.SCIRXST.bit.FE) || (SciaRegs.SCIRXST.bit.PE) || (SciaRegs.SCIFFRX.bit.RXFFOVF) || (SciaRegs.SCIRXST.bit.BRKDT))



402		{



403			return FRAMERR;					/* Providing FRAME error for frame error, parity error and FIFO overflow



404											  as the RTIOStream doesnt have capability to differentitate between errors */



405		}



406



407



408	}



409



410	return NOERROR;



411}

416void scia_xmit_external_mode(char* pmsg, int msglen)



417{



418	int i;



419	for (i = 0; i < msglen; i++) {



420		while (SciaRegs.SCIFFTX.bit.TXFFST == MW_PIL_SCIFIFOLEN) {} /* The buffer is full */



421        #ifdef F2837X_REG_FORMAT



422        SciaRegs.SCITXBUF.bit.TXDT = pmsg[i];



423        #else



424		SciaRegs.SCITXBUF = pmsg[i];



425        #endif



426	}



427



428	while (SciaRegs.SCIFFTX.bit.TXFFST != 0) {}



429



430}

434/* Function: IsEscapeChar16bit ================================================



435 * Abstract:



436 * Check whether the Byte to be transmitted is an escape character, header, or tail



437 *



438 *  Return True on escape character(or header,tail) and byte_select will mention



439 *  the location of the escape character in 16 bits. (Low, High or Both)



440 */



441boolean_T IsEscapeChar16bit(char input, ByteSelect* byte_select)



442{



443



444	if(



445	(input == PACKET_TAIL_16_HL) ||



446	(input == PACKET_HEAD_16_HL) ||



447	(input == ESCAPE_CHARACTER_16_HL) ||



448	(input == MIX_16_HT) ||



449	(input == MIX_16_HE) ||



450	(input == MIX_16_EH) ||



451	(input == MIX_16_ET) ||



452	(input == MIX_16_TH) ||



453	(input == MIX_16_TE)



454



455	)



456	{



457		*byte_select = HIGHLOW_BYTE;



458	 	return true;



459	}



460



461	else if(



462	((input & MASK_HIGHER_BYTE) == PACKET_TAIL_16_L) ||



463	((input & MASK_HIGHER_BYTE) == PACKET_HEAD_16_L) ||



464	((input & MASK_HIGHER_BYTE) == ESCAPE_CHARACTER_16_L)



465	)



466	{



467



468		*byte_select = LOW_BYTE;



469		return true;



470	}



471	else if(



472	((input & MASK_LOWER_BYTE)  == PACKET_TAIL_16_H) ||



473	((input & MASK_LOWER_BYTE)  == PACKET_HEAD_16_H) ||



474	((input & MASK_LOWER_BYTE)  == ESCAPE_CHARACTER_16_H)



475	)



476	{



477		*byte_select = HIGH_BYTE;



478    	return true;



479	}



480



481	 return false;



482}

485int byteswap_L8exp_esc(char* outdata, char* inport, int inportWidth, int typeLen, int* escapecharAdjust)



486{



487	int i, j;



488	int k = 0;



489	int numWrd = (inportWidth * typeLen) / 2; /* number of words (16 bit length) to send */



490	ByteSelect byte_select= LOW_BYTE;



491



492	/* Little Endian, 8bit swap */



493	for (i = 0; i < numWrd; i++) {



494		for (j = 0; j < 2; j++) {



495			outdata[k++] = (inport[i] >> (8*j)) & 255;



496		}



497	}



498



499	return 0;



500}



501#else



502int byteswap_L8exp_esc(char* outdata, char* inport, int inportWidth, int typeLen, int* escapecharAdjust)



503{



504	int i, j;



505	int k = 0;



506	int numWrd = (inportWidth * typeLen) / 2; /* number of words (16 bit length) to send */



507	ByteSelect byte_select= LOW_BYTE;



508



509	/* Little Endian, 8bit swap */



510	for (i = 0; i < numWrd; i++) {



511



512	if(IsEscapeChar16bit(inport[i], &byte_select))



513	{



514		switch(byte_select)



515	 		{



516



517				case LOW_BYTE:



518				outdata[k++] = escape_character;



519				outdata[k++] = ( (inport[i] & 0xff) ) ^ mask_character;



520				outdata[k++] = ( (inport[i] & 0xff00)>>8 );



521				*escapecharAdjust += 1;	//Adjusting the return size



522				break;



523



524				case HIGH_BYTE:



525				outdata[k++] = (inport[i]&0xff);



526				outdata[k++] = escape_character;



527				outdata[k++] = ( (inport[i] & 0xff00)>>8) ^ mask_character;



528				*escapecharAdjust += 1;	//Adjusting the return size



529				break;



530



531				case HIGHLOW_BYTE:



532				outdata[k++] = escape_character;



533				outdata[k++] = ( (inport[i] & 0xff) ) ^ mask_character;



534				outdata[k++] = escape_character;



535				outdata[k++] = ( (inport[i] & 0xff00)>>8) ^ mask_character;



536				*escapecharAdjust += 2;	//Adjusting the return size



537				break;



538



539				default:



540				outdata[k++] = escape_character;



541				outdata[k++] = ( (inport[i] & 0xff) ) ^ mask_character;



542				outdata[k++] = ( (inport[i] & 0xff00)>>8 );



543				*escapecharAdjust += 1;	//Adjusting the return size



544				//default is LOW_BYTE



545



546



547			}



548



549



550	}



551	else



552	{



553		for (j = 0; j < 2; j++) {



554



555



556			outdata[k++] = (inport[i] >> (8*j)) & 255;



557



558



559		}



560



561	}



562



563	}



564



565	return 0;



566}
