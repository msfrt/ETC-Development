9void config_ADCB_SOC0()



10{



11  EALLOW;



12  AdcbRegs.ADCSOC0CTL.bit.CHSEL = 2;   /* Set SOC0 channel select to ADCIN2*/



13  AdcbRegs.ADCSOC0CTL.bit.TRIGSEL = 0;



14  AdcbRegs.ADCSOC0CTL.bit.ACQPS = 14;



15                                 /* Set SOC0 S/H Window to 15 ADC Clock Cycles*/



16  AdcbRegs.ADCINTSOCSEL1.bit.SOC0 = 0;



17                                   /* SOCx No ADCINT Interrupt Trigger Select.*/



18  AdcbRegs.ADCOFFTRIM.bit.OFFTRIM = AdcbRegs.ADCOFFTRIM.bit.OFFTRIM;/* Set Offset Error Correctino Value*/



19  AdcbRegs.ADCCTL1.bit.INTPULSEPOS = 1;



20                                /* Late interrupt pulse trips AdcResults latch*/



21  AdcbRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 0;/* All in round robin mode SOC Priority*/



22  EDIS;



23}

25void InitAdcB()



26{



27  EALLOW;



28  CpuSysRegs.PCLKCR13.bit.ADC_B = 1;



29  AdcbRegs.ADCCTL2.bit.PRESCALE = 8;



30  AdcSetMode(ADC_ADCB, ADC_RESOLUTION_12BIT, ADC_SIGNALMODE_SINGLE);



31



32  //power up the ADC



33  AdcbRegs.ADCCTL1.bit.ADCPWDNZ = 1;



34



35  //delay for 1ms to allow ADC time to power up



36  DELAY_US(1000);



37  EDIS;



38}
